## TASK 1 — CMI Payment Integration (P0)

**Goal:** Wire CMI payment gateway using the existing TypeScript npm package so clinics can collect online payments from Moroccan patients.

### Install

### Create `src/lib/cmi.ts`
```typescript
import CmiClient from 'cmi-payment-nodejs';

interface CmiPaymentParams {
  amount: number;       // in MAD, e.g. 150.00
  orderId: string;      // unique per transaction
  email: string;
  patientName: string;
  clinicId: string;
  clinicName: string;
}

export function createCmiPaymentData(params: CmiPaymentParams) {
  const client = new CmiClient({
    storekey:    process.env.CMI_STORE_KEY!,
    clientid:    process.env.CMI_CLIENT_ID!,
    oid:         `${params.clinicId}-${params.orderId}`,
    shopurl:     process.env.NEXT_PUBLIC_SITE_URL!,
    okUrl:       `${process.env.NEXT_PUBLIC_SITE_URL}/api/payments/cmi/success`,
    failUrl:     `${process.env.NEXT_PUBLIC_SITE_URL}/api/payments/cmi/fail`,
    CallbackURL: `${process.env.NEXT_PUBLIC_SITE_URL}/api/payments/cmi/callback`,
    email:       params.email,
    BillToName:  params.patientName,
    amount:      params.amount.toFixed(2),
    currency:    '504',    // MAD ISO 4217
    lang:        'fr',
    hashAlgorithm: 'ver3',
  });
  return client.getFormData(); // returns { url, fields } for a POST form
}
```

### Create `src/app/api/payments/cmi/callback/route.ts`
- Parse POST body from CMI (url-encoded form)
- Recompute HMAC-SHA512 using `CMI_STORE_KEY` over the exact field list CMI specifies
- If hash matches AND `Response === 'Approved'`: update `appointments.payment_status = 'paid'` for the `oid`
- Return plain text `'ACTION=POSTAUTH'` on success, `'ACTION=FAIL'` on failure — CMI requires this exact response format
- Reference for hash computation: `https://github.com/mehdirochdi/cmi-payment-php/blob/master/src/CMI.php` → `validateCallback()` method

### Create Edge Function `supabase/functions/send-appointment-reminders/index.ts`
```typescript
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2';

const supabase = createClient(
  Deno.env.get('SUPABASE_URL')!,
  Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
);

const REMINDER_WINDOWS = [
  { type: '24h',   minutesBefore: 24 * 60,  template: 'appointment_reminder_24h'  },
  { type: '2h',    minutesBefore: 2 * 60,   template: 'appointment_reminder_2h'   },
  { type: '15min', minutesBefore: 15,        template: 'appointment_reminder_15min' },
];

Deno.serve(async () => {
  const now = new Date();

  for (const window of REMINDER_WINDOWS) {
    const targetTime = new Date(now.getTime() + window.minutesBefore * 60_000);
    const rangeStart = new Date(targetTime.getTime() - 7 * 60_000);  // ±7 min window
    const rangeEnd   = new Date(targetTime.getTime() + 7 * 60_000);

    // Find appointments in the target window that haven't been reminded yet
    const { data: appointments } = await supabase
      .from('appointments')
      .select(`
        id, clinic_id, scheduled_at, status,
        patient:patients(name, phone),
        doctor:users!doctor_id(name),
        clinic:clinics(name, whatsapp_phone_id, whatsapp_access_token)
      `)
      .eq('status', 'confirmed')
      .gte('scheduled_at', rangeStart.toISOString())
      .lte('scheduled_at', rangeEnd.toISOString())
      .not('id', 'in', `(
        select appointment_id from appointment_reminders
        where reminder_type = '${window.type}'
      )`);

    if (!appointments?.length) continue;

    for (const appt of appointments) {
      await sendWhatsAppReminder(appt, window.template);

      await supabase.from('appointment_reminders').insert({
        appointment_id: appt.id,
        clinic_id:      appt.clinic_id,
        reminder_type:  window.type,
      });
    }
  }

  return new Response('ok');
});

async function sendWhatsAppReminder(appt: any, templateName: string) {
  const { clinic, patient, doctor } = appt;
  const date = new Date(appt.scheduled_at);

  await fetch(
    `https://graph.facebook.com/v19.0/${clinic.whatsapp_phone_id}/messages`,
    {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${clinic.whatsapp_access_token}`,
      },
      body: JSON.stringify({
        messaging_product: 'whatsapp',
        to: patient.phone,
        type: 'template',
        template: {
          name: templateName,
          language: { code: 'fr' },
          components: [{
            type: 'body',
            parameters: [
              { type: 'text', text: patient.name },
              { type: 'text', text: doctor.name },
              { type: 'text', text: date.toLocaleDateString('fr-MA') },
              { type: 'text', text: date.toLocaleTimeString('fr-MA', { hour: '2-digit', minute: '2-digit' }) },
              { type: 'text', text: clinic.name },
            ],
          }],
        },
      }),
    }
  );
}
```

```typescript
export async function promoteWaitlist({ doctorId, clinicId, cancelledSlot }) {
  const supabase = createClient(); // server client

  const { data: next } = await supabase
    .from('waitlist')
    .select('*, patient:patients(name, phone)')
    .eq('doctor_id', doctorId)
    .eq('clinic_id', clinicId)
    .is('notified_at', null)
    .order('created_at')
    .limit(1)
    .maybeSingle();

  if (!next) return;

  const expiresAt = new Date(Date.now() + 2 * 60 * 60 * 1000); // 2h to claim

  await supabase.from('waitlist')
    .update({ notified_at: new Date().toISOString(), expires_at: expiresAt.toISOString() })
    .eq('id', next.id);

  // Send WhatsApp with a claim link
  await sendWhatsApp(next.patient.phone, 'slot_available_template', {
    patientName: next.patient.name,
    slotDateTime: cancelledSlot,
    claimUrl: `${process.env.NEXT_PUBLIC_SITE_URL}/booking/claim/${next.id}`,
    expiresIn: '2 heures',
  });
}
```

```typescript
import { Document, Page, Text, View, StyleSheet, pdf } from '@react-pdf/renderer';

const styles = StyleSheet.create({
  page:    { padding: 40, fontSize: 11, fontFamily: 'Helvetica' },
  header:  { marginBottom: 20, borderBottom: '1pt solid #ccc', paddingBottom: 10 },
  title:   { fontSize: 16, fontWeight: 'bold', marginBottom: 4 },
  section: { marginTop: 16 },
  med:     { flexDirection: 'row', marginBottom: 8, gap: 8 },
  label:   { fontWeight: 'bold', width: 120 },
});

export async function generatePrescriptionPDF({ doctor, patient, clinic, medications, notes, date }) {
  const doc = (
    <Document>
      <Page size="A5" style={styles.page}>
        <View style={styles.header}>
          <Text style={styles.title}>Dr. {doctor.name}</Text>
          <Text>{doctor.speciality}</Text>
          <Text>{clinic.address}</Text>
          <Text>Tél: {clinic.phone}</Text>
        </View>
        <View>
          <Text>Patient: {patient.name}</Text>
          <Text>Date: {new Date(date).toLocaleDateString('fr-MA')}</Text>
        </View>
        <View style={styles.section}>
          <Text style={{ fontWeight: 'bold', marginBottom: 8 }}>Ordonnance</Text>
          {medications.map((med, i) => (
            <View key={i} style={styles.med}>
              <Text style={styles.label}>{med.name}</Text>
              <Text>{med.dose} — {med.duration} — {med.instructions}</Text>
            </View>
          ))}
        </View>
        {notes && <View style={styles.section}><Text>{notes}</Text></View>}
      </Page>
    </Document>
  );

  return pdf(doc).toBuffer(); // returns Buffer — upload to R2 or send directly
}
```
